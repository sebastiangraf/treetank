22  /**
23   * Combined Storage with a BDB in the front and a JClouds in the back.
24   * 
25   * @author Sebastian Graf, University of Konstanz
26   * 
27   */
28  public class CombinedStorage implements IBackend {
29  
30      /** Local Backend. */
31      private final BerkeleyStorage mLocalBackend;
32  
33      /** Remote Backend. */
34      private final JCloudsStorage mRemoteBackend;
35  
36      /**
37       * Simple constructor.
38       * 
39       * @param pProperties
40       *            not only the file associated with the database
41       * @param pDataFac
42       *            factory for the datas
43       * @param pMetaFac
44       *            factory for meta bucket
45       * @param pByteHandler
46       *            handling any bytes
47       * @throws TTIOException
48       *             of something odd happens while database-connection
49       */
50      @Inject
51      public CombinedStorage(@Assisted Properties pProperties, IDataFactory pDataFac,
52          IMetaEntryFactory pMetaFac, IByteHandlerPipeline pByteHandler) throws TTIOException {
53          mLocalBackend = new BerkeleyStorage(pProperties, pDataFac, pMetaFac, pByteHandler);
54          mRemoteBackend = new JCloudsStorage(pProperties, pDataFac, pMetaFac, pByteHandler.clone());
55      }
56  
57      /**
58       * {@inheritDoc}
59       */
60      @Override
61      public IBackendWriter getWriter() throws TTException {
62          final IBackendWriter firstWriter = mLocalBackend.getWriter();
63          final IBackendWriter secondWriter = mRemoteBackend.getWriter();
64          return new CombinedWriter(firstWriter, secondWriter);
65      }
66  
67      /**
68       * {@inheritDoc}
69       */
70      @Override
71      public IBackendReader getReader() throws TTException {
72          final IBackendReader firstReader = mLocalBackend.getReader();
73          final IBackendReader secondReader = mRemoteBackend.getReader();
74          return new CombinedReader(firstReader, secondReader);
75      }
76  
77      /**
78       * {@inheritDoc}
79       */
80      @Override
81      public void close() throws TTException {
82          mRemoteBackend.close();
83          mLocalBackend.close();
84      }
85  
86      /**
87       * {@inheritDoc}
88       */
89      @Override
90      public IByteHandlerPipeline getByteHandler() {
91          return mLocalBackend.getByteHandler();
92      }
93  
94      /**
95       * {@inheritDoc}
96       */
97      @Override
98      public boolean truncate() throws TTException {
99          return mLocalBackend.truncate() && mRemoteBackend.truncate();
100     }
101 
102     /**
103      * {@inheritDoc}
104      */
105     @Override
106     public void initialize() throws TTIOException {
107         mRemoteBackend.initialize();
108         mLocalBackend.initialize();
109     }
