41  /**
42   * <h1>Database Configuration</h1>
43   * 
44   * <p>
45   * Represents a configuration of a database. Includes all settings which have to be made when it comes to the
46   * creation of the database. Since the settings are persisted after creation, it must contain a link to the
47   * file defined by the interface {@link IConfigureSerializable}.
48   * </p>
49   * 
50   * @author Sebastian Graf, University of Konstanz
51   */
52  public final class DatabaseConfiguration {
53  
54      /**
55       * Paths for a {@link org.treetank.access.Database}. Each {@link org.treetank.access.Database} has the
56       * same folder.layout.
57       */
58      public enum Paths {
59  
60          /** File to store db settings. */
61          ConfigBinary(new File("dbsetting.obj"), false),
62          /** File to store the data. */
63          Data(new File("resources"), true);
64  
65          /** Location of the file. */
66          private final File mFile;
67  
68          /** Is the location a folder or no? */
69          private final boolean mIsFolder;
70  
71          /**
72           * Constructor.
73           * 
74           * @param pFile
75           *            to be set
76           * @param pIsFolder
77           *            to be set.
78           */
79          private Paths(final File pFile, final boolean pIsFolder) {
80              this.mFile = pFile;
81              this.mIsFolder = pIsFolder;
82          }
83  
84          /**
85           * Getting the file for the kind.
86           * 
87           * @return the file to the kind
88           */
89          public File getFile() {
90              return mFile;
91          }
92  
93          /**
94           * Check if file is denoted as folder or not.
95           * 
96           * @return boolean if file is folder
97           */
98          public boolean isFolder() {
99              return mIsFolder;
100         }
101 
102         /**
103          * Checking a structure in a folder to be equal with the data in this
104          * enum.
105          * 
106          * @param pFile
107          *            to be checked
108          * @return -1 if less folders are there, 0 if the structure is equal to
109          *         the one expected, 1 if the structure has more folders
110          */
111         public static int compareStructure(final File pFile) {
112             int existing = 0;
113             for (final Paths paths : values()) {
114                 final File currentFile = new File(pFile, paths.getFile().getName());
115                 if (currentFile.exists()) {
116                     existing++;
117                 }
118             }
119             return existing - values().length;
120         }
121 
122         /**
123          * Deleting a storage recursive. Used for deleting a databases
124          * 
125          * @param pFile
126          *            which should be deleted included descendants
127          * @return true if delete is valid
128          */
129         public static boolean recursiveDelete(final File pFile) {
130             if (pFile.isDirectory()) {
131                 for (final File child : pFile.listFiles()) {
132                     if (!recursiveDelete(child)) {
133                         return false;
134                     }
135                 }
136             }
137             return pFile.delete();
138         }
139 
140     }
141 
142     /** Path to file. */
143     public final File mFile;
144 
145     /**
146      * Constructor with the path to be set.
147      * 
148      * @param paramFile
149      *            file to be set
150      */
151     public DatabaseConfiguration(final File paramFile) {
152         mFile = paramFile;
153     }
154 
155     /**
156      * {@inheritDoc}
157      */
158     @Override
159     public String toString() {
160         StringBuilder builder = new StringBuilder();
161         builder.append("DatabaseConfiguration [mFile=");
162         builder.append(mFile);
163         builder.append("]");
164         return builder.toString();
165     }
166 
167     /**
168      * {@inheritDoc}
169      */
170     @Override
171     public boolean equals(final Object pObj) {
172         return this.hashCode() == pObj.hashCode();
173     }
174 
175     /**
176      * {@inheritDoc}
177      */
178     @Override
179     public int hashCode() {
180         final int prime = 72277;
181         int result = 13;
182         result = prime * result + mFile.hashCode();
183         return result;
184     }
185 
186     /**
187      * Serializing a {@link DatabaseConfiguration} to a json file.
188      * 
189      * @param pConfig
190      *            to be serialized
191      * @throws TTIOException
192      */
193     public static void serialize(final DatabaseConfiguration pConfig) throws TTIOException {
194         try {
195             FileWriter fileWriter =
196                 new FileWriter(new File(pConfig.mFile, Paths.ConfigBinary.getFile().getName()));
197             JsonWriter jsonWriter = new JsonWriter(fileWriter);
198             jsonWriter.beginObject();
199             jsonWriter.name("file").value(pConfig.mFile.getAbsolutePath());
200             jsonWriter.endObject();
201             jsonWriter.close();
202             fileWriter.close();
203         } catch (FileNotFoundException fileExec) {
204             throw new TTIOException(fileExec);
205         } catch (IOException ioexc) {
206             throw new TTIOException(ioexc);
207         }
208     }
209 
210     /**
211      * Generate a DatabaseConfiguration out of a file.
212      * 
213      * @param pFile
214      *            where the DatabaseConfiguration lies in as json
215      * @return a new {@link DatabaseConfiguration} class
216      * @throws TTIOException
217      */
218     public static DatabaseConfiguration deserialize(final File pFile) throws TTIOException {
219         try {
220             FileReader fileReader = new FileReader(new File(pFile, Paths.ConfigBinary.getFile().getName()));
221             JsonReader jsonReader = new JsonReader(fileReader);
222             jsonReader.beginObject();
223             assert jsonReader.nextName().equals("file");
224             File file = new File(jsonReader.nextString());
225             jsonReader.endObject();
226             jsonReader.close();
227             fileReader.close();
228             return new DatabaseConfiguration(file);
229         } catch (FileNotFoundException fileExec) {
230             throw new TTIOException(fileExec);
231         } catch (IOException ioexc) {
232             throw new TTIOException(ioexc);
233         }
234     }
