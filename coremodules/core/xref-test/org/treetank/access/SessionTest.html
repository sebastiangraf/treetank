73              mResourceConfig.create(StandardSettings.getProps(CoreTestHelper.PATHS.PATH1.getFile()
74                  .getAbsolutePath(), resource2));

76          assertTrue(mHolder.getStorage().createResource(config1));
77          assertTrue(mHolder.getStorage().createResource(config2));

79          final ISession session1 =
80              mHolder.getStorage().getSession(new SessionConfiguration(resource1, StandardSettings.KEY));
81          final ISession session2 =
82              mHolder.getStorage().getSession(new SessionConfiguration(resource2, StandardSettings.KEY));
83

85          IBucketWriteTrx wtx2 = session2.beginBucketWtx();
