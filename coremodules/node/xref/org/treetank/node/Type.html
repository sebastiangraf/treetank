904     /**
905      * Returns the least common base type of the given types.
906      * 
907      * @param mType1
908      *            First type
909      * @param mType2
910      *            Second type
911      * @return the common type of the parameter types
912      */
913     public static Type getLeastCommonType(final Type mType1, final Type mType2) {
914 
915         Type a = mType1;
916         Type b = mType2;
917 
918         if (a.mDerivedFrom != null && b.mDerivedFrom != null) {
919 
920             // to get a common type, we need both type and baseTypes
921             // respectively on
922             // the same level of precedence
923             if (a.mPrecedence != b.mPrecedence) {
924                 while (a.mPrecedence > b.mPrecedence) {
925                     a = a.mDerivedFrom;
926                 }
927 
928                 while (a.mPrecedence < b.mPrecedence) {
929                     b = b.mDerivedFrom;
930                 }
931             }
932 
933             while (a != b) {
934                 assert a.mPrecedence == b.mPrecedence;
935                 // step up the basetypes of the parameters until a common type
936                 // is found
937                 // since all types are derived from the same base type
938                 // (anyType), this
939                 // is not an infinite loop, in fact this loop will be left when
940                 // a's and
941                 // b's type is 'anyType', at the latest.
942                 a = a.mDerivedFrom;
943                 b = b.mDerivedFrom;
944             }
945             assert a == b;
946 
947         } else {
948             assert a == ANY_TYPE || b == ANY_TYPE;
949 
950             a = ANY_TYPE;
951         }
952 
953         return a;
954     }

956     /**
957      * Tests whether a type is derived by restriction from a certain type.
958      * 
959      * @param mExpectedType
960      *            the type to check, if this type is derived from
961      * @return true, if this type is derived from the input type.
962      */
963     public boolean derivesFrom(final Type mExpectedType) {
964 
965         if (this == mExpectedType) {
966             return true;
967         }
968 
969         if (this.mPrecedence > mExpectedType.mPrecedence) {
970             // only if actualType has a lower precedence than expectedTyte,
971             // expectedTyte can be a baseType of actualType
972             Type baseType = this.mDerivedFrom;
973             while (baseType.mPrecedence > mExpectedType.mPrecedence) {
974                 baseType = baseType.mDerivedFrom;
975             }
976 
977             assert baseType.mPrecedence == mExpectedType.mPrecedence;
978             return baseType == mExpectedType;
979 
980         } else {
981             // if actualType has higher or equal precedence than expectedTyte,
982             // it can
983             // not be derived from expectedTyte, instead, it could be even the
984             // other
985             // way around
986             return false;
987         }
988     }

1002      * Determines, whether a value of the current type can be casted to the
1003      * input type, as defined in XML Schema 1.1-1 Spec.
1004      * 
1005      * @param mTargetType
1006      *            type to which the current value shall be casted to
1007      * @param mValue
1008      *            value of the source value
1009      * @return true, if the source value is castable to the target type.
1010      * @throws TTXPathException
1011      *             if casts fails
1012      */
1013     public boolean isCastableTo(final Type mTargetType, final String mValue) throws TTXPathException {
1014 
1015         // (4.a) a type can always be casted to itself, or one of its super
1016         // types
1017         if (this == mTargetType || derivesFrom(mTargetType)) {
1018             return true;
1019         }
1020 
1021         if (isPrimitive() && mTargetType.isPrimitive()) {
1022             // source and target types are both primitive atomic types
1023 
1024             if (mTargetType == UNTYPED_ATOMIC || mTargetType == STRING || castsTo(mTargetType)
1025                 || mTargetType.facetIsSatisfiedBy(mValue)) {
1026                 return true;
1027             }
1028         }
1029 
1030         if (!mTargetType.isPrimitive() && mTargetType.derivesFrom(this)) {
1031             // (4.d) if target type is non-primitive-atomic-type and target type
1032             // is supertype of source type, the source type must satisfy all
1033             // facets
1034             // of the target type.
1035             return mTargetType.facetIsSatisfiedBy(mValue);
1036 
1037         } else {
1038             // (4.e) if the primitive base type of the source is castable to the
1039             // primitive base type of the target type, the source is castable to
1040             // the target, if the facets of the target is satisfied.
1041             // getPrimitiveBaseType();
1042             // TODO: what is that??
1043 
1044             return (getPrimitiveBaseType().isCastableTo(mTargetType.getPrimitiveBaseType(), mValue) && mTargetType
1045                 .facetIsSatisfiedBy(mValue));
1046         }
1047 
1048         // throw new XPathError(ErrorType.XPTY0004);
1049 
1050     }
