165     @Test
166     public void testGetBaseURI() throws Exception {
167         // Test with xml:base specified.
168         final File source =
169             new File("src" + File.separator + "test" + File.separator + "resources" + File.separator + "data"
170                 + File.separator + "testBaseURI.xml");
171 
172         final StorageConfiguration db2 = new StorageConfiguration(TestHelper.PATHS.PATH2.getFile());
173 
174         Storage.createStorage(db2);
175         final IStorage storage = Storage.openStorage(TestHelper.PATHS.PATH2.getFile());
176         Properties props = new Properties();
177         props.setProperty(org.treetank.io.IConstants.DBFILE, TestHelper.PATHS.PATH2.getFile().getAbsolutePath());
178         props.setProperty(org.treetank.io.IConstants.RESOURCE, TestHelper.RESOURCENAME);
179         props.setProperty(FilesystemConstants.PROPERTY_BASEDIR, new File(new File(new File(
180             props.getProperty(IConstants.DBFILE), StorageConfiguration.Paths.Data.getFile()
181                 .getName()), props.getProperty(IConstants.RESOURCE)),
182             ResourceConfiguration.Paths.Data.getFile().getName()).getAbsolutePath());
183         props.setProperty(Constants.PROPERTY_CREDENTIAL, "test");
184         props.setProperty(IConstants.JCLOUDSTYPE, "filesystem");
185         storage.createResource(mResourceConfig.create(props, 1));
186         final ISession session =
187             storage.getSession(new SessionConfiguration(TestHelper.RESOURCENAME, StandardSettings.KEY));
188         final INodeWriteTrx wtx =
189             new NodeWriteTrx(session, session.beginPageWriteTransaction(), HashKind.Rolling);
190         SaxonHelper.createDocumentRootNode(wtx);
191         final XMLEventReader reader = XMLShredder.createFileReader(source);
192         final XMLShredder shredder = new XMLShredder(wtx, reader, EShredderInsert.ADDASFIRSTCHILD);
193         shredder.call();
194         wtx.close();
195 
196         final Processor proc = new Processor(false);
197         final NodeInfo doc = new DocumentWrapper(session, proc.getUnderlyingConfiguration());
198 
199         doc.getNamePool().allocate("xml", "http://www.w3.org/XML/1998/namespace", "base");
200         doc.getNamePool().allocate("", "", "baz");
201 
202         final NameTest test = new NameTest(Type.ELEMENT, "", "baz", doc.getNamePool());
203         final AxisIterator iterator = doc.iterateAxis(Axis.DESCENDANT, test);
204         final NodeInfo baz = (NodeInfo)iterator.next();
205 
206         assertEquals("http://example.org", baz.getBaseURI());
207         session.close();
208         storage.close();
209 
210     }

241     @Test
242     public void testGetStringValueCS() {
243         // Test on document node.
244         assertEquals("oops1foooops2baroops3", node.getStringValueCS());
245 
246         // Test on element node.
247         AxisIterator iterator = node.iterateAxis(Axis.DESCENDANT);
248         node = (NodeWrapper)iterator.next();
249         assertEquals("oops1foooops2baroops3", node.getStringValueCS());
250 
251         // Test on namespace node.
252         iterator = node.iterateAxis(Axis.NAMESPACE);
253         NamespaceNodeImpl namespace = (NamespaceNodeImpl)iterator.next();
254 
255         /*
256          * Elements have always the default xml:NamespaceConstant.XML namespace,
257          * so we have to search if "ns" is found somewhere in the iterator
258          * (order unpredictable because it's implemented with a HashMap
259          * internally).
260          */
261         while (!"ns".equals(namespace.getStringValueCS()) && namespace != null) {
262             namespace = (NamespaceNodeImpl)iterator.next();
263         }
264 
265         if (namespace == null) {
266             Assert.fail("namespace is null!");
267         } else {
268             assertEquals("ns", namespace.getStringValueCS());
269         }
270 
271         // Test on attribute node.
272         final NodeWrapper attrib = (NodeWrapper)node.iterateAxis(Axis.ATTRIBUTE).next();
273         assertEquals("j", attrib.getStringValueCS());
274 
275         // Test on text node.
276         final NodeWrapper text = (NodeWrapper)node.iterateAxis(Axis.CHILD).next();
277         assertEquals("oops1", text.getStringValueCS());
278     }
