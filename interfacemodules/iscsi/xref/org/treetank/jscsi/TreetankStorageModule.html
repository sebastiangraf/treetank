100     /**
101      * Creates a new {@link TreetankStorageModule} backed by the specified {@link IStorage}.
102      * 
103      * @param sizeInBlocks
104      *            blocksize for this module
105      * @param conf
106      *            the fully initialized {@link StorageConfiguration}
107      * @throws TTException
108      */
109     public TreetankStorageModule(final long sizeInBlocks, final int blockSize,
110         final StorageConfiguration conf, final File file) throws TTException {
111 
112         LOGGER.info("Initializing storagemodule with: sizeInBlocks=" + sizeInBlocks + ", blockSize="
113             + blockSize);
114 
115         this.sizeInBlocks = sizeInBlocks;
116         this.blockSize = blockSize;
117 
118         Injector injector =
119             Guice.createInjector(new ModuleSetter().setNodeFacClass(ByteNodeFactory.class).setMetaFacClass(
120                 ISCSIMetaPageFactory.class).createModule());
121         IBackendFactory backend = injector.getInstance(IBackendFactory.class);
122         IRevisioning revision = injector.getInstance(IRevisioning.class);
123 
124         // Creating and opening the storage.
125         // Making it ready for usage.
126         if (!file.exists()) {
127             Storage.truncateStorage(conf);
128             Storage.createStorage(conf);
129         }
130 
131         storage = Storage.openStorage(file);
132 
133         Properties props = StandardSettings.getProps(file.getAbsolutePath(), "jscsi-target");
134         ResourceConfiguration mResourceConfig =
135             new ResourceConfiguration(props, backend, revision, new ByteNodeFactory(),
136                 new ISCSIMetaPageFactory());
137         storage.createResource(mResourceConfig);
138 
139         session = storage.getSession(new SessionConfiguration("jscsi-target", null));
140         mRtx = new IscsiWriteTrx(session.beginPageWriteTransaction(), session);
141 
142         try {
143             createStorage();
144         } catch (IOException e) {
145             // TODO Auto-generated catch block
146             e.printStackTrace();
147         }
148     }

150     private void createStorage() throws IOException {
151 
152         LOGGER.info("Creating storage with " + sizeInBlocks + " blocks of size 512.");
153 
154         try {
155 
156             INode node = this.mRtx.getCurrentNode();
157 
158             if (node != null)
159                 return;
160             for (int i = 0; i < sizeInBlocks; i++) {
161                 try {
162                     this.mRtx.bootstrap(new byte[(int)(blockSize)]);
163                 } catch (TTException e) {
164                     throw new IOException("The creation of a new node was started and somehow didn't finish.");
165                 }
166             }
167 
168             this.mRtx.commit();
169 
170         } catch (TTException e) {
171             // TODO Auto-generated catch block
172             e.printStackTrace();
173         }
174 
175     }

180     public int checkBounds(long logicalBlockAddress, int transferLengthInBlocks) {
181 
182         if (logicalBlockAddress < 0 || logicalBlockAddress >= sizeInBlocks)
183             return 1;
184         if (transferLengthInBlocks < 0 || logicalBlockAddress + transferLengthInBlocks > sizeInBlocks)
185             return 2;
186         return 0;
187     }

200     public void read(byte[] bytes, int bytesOffset, int length, long storageIndex) throws IOException {
201 
202         LOGGER.info("Starting to read with param: \nbytesOffset = " + bytesOffset + "\nlength = " + length
203             + "\nstorageIndex = " + storageIndex);
204         // Using the most recent revision
205         if (bytesOffset + length > bytes.length) {
206             throw new IOException();
207         }
208         int realIndex = (int)(storageIndex / blockSize);
209         LOGGER.info("Starting to read realIndex " + realIndex);
210 
211         ByteArrayDataOutput output = ByteStreams.newDataOutput(length);
212 
213         int inc = (length % blockSize == 0) ? 0 : 1;
214         for (long i = realIndex; i < ((length + storageIndex) / blockSize) + inc; i++) {
215             setCursorToIndex(i);
216 
217             INode node = this.mRtx.getCurrentNode();
218 
219             byte[] val = ((ByteNode)node).getVal();
220 
221             output.write(val);
222 
223         }
224 
225         System.arraycopy(output.toByteArray(), 0, bytes, bytesOffset, length);
226 
227         LOGGER.info("Giving back bytes: " + Arrays.toString(bytes).substring(0, 1000));
228     }

238             // Using the most recent revision

243             int realIndex = (int)(storageIndex / blockSize);
244 
245             ByteArrayDataOutput output = ByteStreams.newDataOutput(length);
246             System.out.println(bytes.length - (blockSize - (length % blockSize)));
247             output.write(bytes, bytesOffset, length);
248 
249             ByteArrayDataInput input = ByteStreams.newDataInput(output.toByteArray());
250 
251             LOGGER.info("Writing from node " + realIndex + " to node "
252                 + (((length + storageIndex) / blockSize)));
253 
254             for (long i = realIndex; i < ((length + storageIndex) / blockSize); i++) {
255                 setCursorToIndex(i);
256                 byte[] val = new byte[blockSize];
257 
258                 input.readFully(val);
259 
260                 this.mRtx.setValue(val);
261             }
262 
263             if (length % blockSize != 0) {
264                 setCursorToIndex((int)((length + storageIndex) / blockSize));
265                 INode node = this.mRtx.getCurrentNode();
266 
267                 byte[] val = ((ByteNode)node).getVal();
268 
269                 input.readFully(val, 0, (length % blockSize));
270 
271                 this.mRtx.setValue(val);
272             }
273 
274             this.mRtx.commit();
275         } catch (TTException e) {
276             throw new IOException(e.getMessage());
277         }

302     private void setCursorToIndex(long pIndex) throws IOException {
303 
304         ByteNode node = (ByteNode)this.mRtx.getCurrentNode();
305 
306         if (node == null) {
307             /*
308              * Only used for lazy creation. ByteNode newNode = new ByteNode(0, new
309              * byte[(int) (blockSize)]); newNode.setIndex(0); try { ((IIscsiWriteTrx)
310              * iWtx).insert(newNode); return; } catch (TTException e) {
311              */
312             throw new IOException("The creation of a new node was started and somehow didn't finish.");
313             // }
314         }
315 
316         if (node.getIndex() == pIndex) {
317             return;
318         } else if (node.getIndex() < pIndex) {
319             do {
320                 this.mRtx.nextNode();
321                 node = (ByteNode)this.mRtx.getCurrentNode();
322                 if (node.getIndex() == pIndex) {
323                     return;
324                 }
325             } while (node.getNextNodeKey() != 0);
326         } else {
327             do {
328                 this.mRtx.previousNode();
329                 node = (ByteNode)this.mRtx.getCurrentNode();
330                 if (node.getIndex() == pIndex) {
331                     return;
332                 }
333             } while (node.getPreviousNodeKey() >= 0);
334         }
335         throw new IOException(toStringHelper("Node never found with index").add("index", pIndex).toString());
336 
337     }
