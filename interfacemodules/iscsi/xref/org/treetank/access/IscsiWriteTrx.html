52      public IscsiWriteTrx(IPageWriteTrx pPageTrx, ISession pSession) throws TTException {
53  
54          mSession = pSession;

58      /**
59       * {@inheritDoc}
60       */
61      @Override
62      public void bootstrap(byte[] bytes) throws TTException {
63          if (mDelegate.getCurrentNode() != null) {
64              ByteNode node = new ByteNode(getPageTransaction().incrementNodeKey(), new byte[512]);
65              node.setVal(bytes);
66              node.setIndex(node.getNodeKey());
67              ByteNode lastNode = (ByteNode)getPageTransaction().getNode(node.getNodeKey() - 1);
68              node.setPreviousNodeKey(lastNode.getNodeKey());
69              lastNode.setNextNodeKey(node.getNodeKey());
70              getPageTransaction().setNode(lastNode);
71              getPageTransaction().setNode(node);
72          } else {
73              ByteNode node = new ByteNode(getPageTransaction().incrementNodeKey(), new byte[512]);
74              node.setIndex(0);
75              getPageTransaction().setNode(node);
76              mDelegate.moveTo(node.getNodeKey());
77          }
78  
79      }

81      /**
82       * {@inheritDoc}
83       */
84      @Override
85      public void insertAfter(byte[] vals) throws TTException {
86  
87          ByteNode node = new ByteNode(getPageTransaction().incrementNodeKey(), new byte[512]);
88          if (((ByteNode)mDelegate.getCurrentNode()).hasNext()) {
89              // Linking the new node.
90              ByteNode nextNode =
91                  (ByteNode)getPageTransaction().getNode(
92                      ((ByteNode)mDelegate.getCurrentNode()).getNextNodeKey());
93              nextNode.setPreviousNodeKey(node.getNodeKey());
94              getPageTransaction().setNode(nextNode);
95  
96              incrementAllFollowingIndizes(mDelegate.getCurrentNode());
97          }
98  
99          node.setNextNodeKey(((ByteNode)mDelegate.getCurrentNode()).getNextNodeKey());
100         node.setPreviousNodeKey(((ByteNode)mDelegate.getCurrentNode()).getNodeKey());
101         node.setIndex(((ByteNode)mDelegate.getCurrentNode()).getIndex() + 1);
102         getPageTransaction().setNode(node);
103 
104         ByteNode currNode = (ByteNode)getPageTransaction().getNode(mDelegate.getCurrentNode().getNodeKey());
105 
106         currNode.setNextNodeKey(node.getNodeKey());
107         getPageTransaction().setNode(currNode);
108 
109     }

111     /**
112      * {@inheritDoc}
113      */
114     @Override
115     public void remove() throws TTException {
116         checkState(!mDelegate.isClosed(), "Transaction is already closed.");
117 
118         if (((ByteNode)mDelegate.getCurrentNode()).hasPrevious()
119             && ((ByteNode)mDelegate.getCurrentNode()).hasNext()) {
120             ByteNode previousNode =
121                 (ByteNode)getPageTransaction().getNode(
122                     ((ByteNode)mDelegate.getCurrentNode()).getPreviousNodeKey());
123 
124             previousNode.setNextNodeKey(((ByteNode)mDelegate.getCurrentNode()).getNextNodeKey());
125 
126             getPageTransaction().setNode(previousNode);
127 
128             ByteNode nextNode =
129                 (ByteNode)getPageTransaction().getNode(
130                     ((ByteNode)mDelegate.getCurrentNode()).getNextNodeKey());
131 
132             nextNode.setPreviousNodeKey(previousNode.getNodeKey());
133 
134             getPageTransaction().setNode(nextNode);
135 
136             long nodeKey = mDelegate.getCurrentNode().getNodeKey();
137             INode deleteNode = new NodePage.DeletedNode(nodeKey);
138             getPageTransaction().setNode(deleteNode);
139 
140             decrementAllFollowingIndizes(previousNode);
141         }
142 
143     }

168     /**
169      * {@inheritDoc}
170      */
171     @Override
172     public void abort() throws TTException {
173         checkState(!mDelegate.isClosed(), "Transaction is already closed.");
174 
175         long revisionToSet = 0;
176         revisionToSet = mDelegate.mPageReadTrx.getRevision() - 1;
177 
178         getPageTransaction().close();
179 
180         // Reset internal transaction state to last committed uber page.
181         mDelegate.setPageTransaction(mSession.beginPageWriteTransaction(revisionToSet));
182 
183     }

269     /**
270      * A help method to increment the indizes following to this node
271      * without altering the index of the given node itself.
272      * 
273      * This is useful when inserting inbetween two nodes
274      * because following indizes have to be altered aswell.
275      * 
276      * @param node
277      * @return true if successful, false otherwise
278      * @throws TTException
279      */
280     private boolean incrementAllFollowingIndizes(INode node) throws TTException {
281 
282         ByteNode nextNode = (ByteNode)node;
283 
284         do {
285             nextNode = (ByteNode)getPageTransaction().getNode(nextNode.getNextNodeKey());
286             nextNode.incIndex();
287             getPageTransaction().setNode(nextNode);
288 
289         } while (nextNode.hasNext());
290 
291         return true;
292     }
293 
294     /**
295      * A help method to decrement the indizes following to this node
296      * without altering the index of the given node itself.
297      * 
298      * This is useful when removing in the middle of the list and the
299      * indizes have to be cut down.
300      * 
301      * @param node
302      * @return true if successful, false otherwise
303      * @throws TTException
304      */
305     private boolean decrementAllFollowingIndizes(INode node) throws TTException {
306 
307         ByteNode nextNode = (ByteNode)node;
308 
309         do {
310             nextNode = (ByteNode)getPageTransaction().getNode(nextNode.getNextNodeKey());
311             nextNode.decIndex();
312             getPageTransaction().setNode(nextNode);
313 
314         } while (nextNode.hasNext());
315 
316         return true;
317     }
