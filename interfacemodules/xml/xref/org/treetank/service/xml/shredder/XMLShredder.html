160         checkNotNull(paramAddAsFirstChild);

162         mWtx = paramWtx;
163         mReader = paramReader;

165         mCommit = paramCommit;
166         mLatch = new CountDownLatch(1);
167         if (mWtx.getNode() == null) {

195     /**
196      * Insert new content based on a StAX parser {@link XMLStreamReader}.
197      * 
198      * @throws TTException
199      *             if something went wrong while inserting
200      */
201     protected final void insertNewContent() throws TTException {
202         try {
203             Stack<Long> leftSiblingKeyStack = new Stack<Long>();
204 
205             leftSiblingKeyStack.push(NULL_NODE);
206             boolean firstElement = true;
207             int level = 0;
208             QName rootElement = null;
209             boolean endElemReached = false;
210             StringBuilder sBuilder = new StringBuilder();
211 
212             // Iterate over all nodes.
213             while (mReader.hasNext() && !endElemReached) {
214                 final XMLEvent event = mReader.nextEvent();
215 
216                 switch (event.getEventType()) {
217                 case XMLStreamConstants.START_ELEMENT:
218                     level++;
219                     leftSiblingKeyStack = addNewElement(leftSiblingKeyStack, (StartElement)event);
220                     if (firstElement) {
221                         firstElement = false;
222                         rootElement = event.asStartElement().getName();
223                     }
224                     break;
225                 case XMLStreamConstants.END_ELEMENT:
226                     level--;
227                     if (level == 0 && rootElement != null
228                         && rootElement.equals(event.asEndElement().getName())) {
229                         endElemReached = true;
230                     }
231                     leftSiblingKeyStack.pop();
232                     mWtx.moveTo(leftSiblingKeyStack.peek());
233                     break;
234                 case XMLStreamConstants.CHARACTERS:
235                     if (mReader.peek().getEventType() == XMLStreamConstants.CHARACTERS) {
236                         sBuilder.append(event.asCharacters().getData().trim());
237                     } else {
238                         sBuilder.append(event.asCharacters().getData().trim());
239                         leftSiblingKeyStack = addNewText(leftSiblingKeyStack, sBuilder.toString());
240                         sBuilder = new StringBuilder();
241                     }
242                     break;
243                 default:
244                     // Node kind not known.
245                 }
246             }
247         } catch (final XMLStreamException e) {
248             throw new TTIOException(e);
249         }
250     }

252     /**
253      * Add a new element node.
254      * 
255      * @param paramLeftSiblingKeyStack
256      *            stack used to determine if the new element has to be inserted
257      *            as a right sibling or as a new child (in the latter case is
258      *            NULL on top of the stack)
259      * @param paramEvent
260      *            the current event from the StAX parser
261      * @return the modified stack
262      * @throws TTException
263      *             if adding {@link ElementNode} fails
264      */
265     protected final Stack<Long> addNewElement(final Stack<Long> paramLeftSiblingKeyStack,
266         final StartElement paramEvent) throws TTException {
267         assert paramLeftSiblingKeyStack != null && paramEvent != null;
268         long key;
269 
270         final QName name = paramEvent.getName();
271 
272         if (mFirstChildAppend == EShredderInsert.ADDASRIGHTSIBLING) {
273             checkState(mWtx.getNode().getKind() != IConstants.ROOT,
274                 "Subtree can not be inserted as sibling of Root");
275             key = mWtx.insertElementAsRightSibling(name);
276             mFirstChildAppend = EShredderInsert.ADDASFIRSTCHILD;
277         } else {
278             if (paramLeftSiblingKeyStack.peek() == NULL_NODE) {
279                 key = mWtx.insertElementAsFirstChild(name);
280             } else {
281                 key = mWtx.insertElementAsRightSibling(name);
282             }
283         }
284 
285         paramLeftSiblingKeyStack.pop();
286         paramLeftSiblingKeyStack.push(key);
287         paramLeftSiblingKeyStack.push(NULL_NODE);
288 
289         // Parse namespaces.
290         for (final Iterator<?> it = paramEvent.getNamespaces(); it.hasNext();) {
291             final Namespace namespace = (Namespace)it.next();
292             mWtx.insertNamespace(new QName(namespace.getNamespaceURI(), "", namespace.getPrefix()));
293             mWtx.moveTo(key);
294         }
295 
296         // Parse attributes.
297         for (final Iterator<?> it = paramEvent.getAttributes(); it.hasNext();) {
298             final Attribute attribute = (Attribute)it.next();
299             mWtx.insertAttribute(attribute.getName(), attribute.getValue());
300             mWtx.moveTo(key);
301         }
302         return paramLeftSiblingKeyStack;
303     }

305     /**
306      * Add a new text node.
307      * 
308      * @param paramLeftSiblingKeyStack
309      *            stack used to determine if the new element has to be inserted
310      *            as a right sibling or as a new child (in the latter case is
311      *            NULL on top of the stack)
312      * @param paramText
313      *            the text string to add
314      * @return the modified stack
315      * @throws TTException
316      *             if adding text fails
317      */
318     protected final Stack<Long>
319         addNewText(final Stack<Long> paramLeftSiblingKeyStack, final String paramText) throws TTException {
320         assert paramLeftSiblingKeyStack != null;
321         final String text = paramText;
322         long key;
323         final ByteBuffer textByteBuffer = ByteBuffer.wrap(TypedValue.getBytes(text));
324         if (textByteBuffer.array().length > 0) {
325 
326             if (paramLeftSiblingKeyStack.peek() == NULL_NODE) {
327                 key = mWtx.insertTextAsFirstChild(new String(textByteBuffer.array()));
328             } else {
329                 key = mWtx.insertTextAsRightSibling(new String(textByteBuffer.array()));
330             }
331 
332             paramLeftSiblingKeyStack.pop();
333             paramLeftSiblingKeyStack.push(key);
334 
335         }
336         return paramLeftSiblingKeyStack;
337     }

339     /**
340      * Main method.
341      * 
342      * @param paramArgs
343      *            input and output files
344      * @throws Exception
345      *             if any exception occurs
346      */
347     public static void main(final String... paramArgs) throws Exception {
348         checkArgument(paramArgs.length == 2, "Usage: XMLShredder input.xml output.tnk");
349 
350         System.out.print("Shredding '" + paramArgs[0] + "' to '" + paramArgs[1] + "' ... ");
351         final long time = System.currentTimeMillis();
352 
353         Injector injector =
354             Guice.createInjector(new ModuleSetter().setDataFacClass(TreeNodeFactory.class).setMetaFacClass(
355                 NodeMetaPageFactory.class).createModule());
356         IBackendFactory storage = injector.getInstance(IBackendFactory.class);
357         IRevisioning revision = injector.getInstance(IRevisioning.class);
358 
359         final File target = new File(paramArgs[1]);
360         final StorageConfiguration config = new StorageConfiguration(target);
361         Storage.truncateStorage(config);
362         Storage.createStorage(config);
363         final IStorage db = Storage.openStorage(target);
364         Properties props = StandardSettings.getProps(target.getAbsolutePath(), "shredded");
365         db.createResource(new ResourceConfiguration(props, storage, revision, new TreeNodeFactory(),
366             new NodeMetaPageFactory()));
367         final ISession session = db.getSession(new SessionConfiguration("shredded", StandardSettings.KEY));
368         final INodeWriteTrx wtx =
369             new NodeWriteTrx(session, session.beginBucketWtx(), HashKind.Rolling);
370         // generating root node
371         final NodeDelegate nodeDel = new NodeDelegate(wtx.getPageWtx().incrementDataKey(), NULL_NODE, 0);
372         DocumentRootNode node =
373             new DocumentRootNode(nodeDel, new StructNodeDelegate(nodeDel, NULL_NODE, NULL_NODE, NULL_NODE, 0));
374         wtx.getPageWtx().setData(node);
375         wtx.moveTo(org.treetank.node.IConstants.ROOT_NODE);
376 
377         final XMLEventReader reader = createFileReader(new File(paramArgs[0]));
378         final XMLShredder shredder = new XMLShredder(wtx, reader, EShredderInsert.ADDASFIRSTCHILD);
379         shredder.call();
380 
381         wtx.close();
382         session.close();
383 
384         System.out.println(" done [" + (System.currentTimeMillis() - time) + "ms].");
385     }
