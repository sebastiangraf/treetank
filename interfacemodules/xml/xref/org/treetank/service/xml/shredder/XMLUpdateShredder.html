263     /**
264      * Invoking the shredder.
265      * 
266      * @throws TTException
267      *             if Treetank encounters something went wrong
268      * @return revision of last revision (before commit)
269      */
270     @Override

272         updateOnly();
273         if (mCommit == EShredderCommit.COMMIT) {

275         }
276         return null;

286         try {
287             // Initialize variables.
288             mLevelInToShredder = 0;
289             // mElemsParsed = 0;
290             // mIsLastNode = false;
291             mMovedToRightSibling = false;
292             boolean firstEvent = true;
293

298             // // executed.

302                 // Find the start key for the update operation.
303                 long startkey = ROOT_NODE + 1;

306                 }
307             }
308 
309             XMLEvent event = null;
310             StringBuilder sBuilder = new StringBuilder();
311             final XMLEventFactory fac = XMLEventFactory.newInstance();
312 
313             // Iterate over all nodes.
314             while (mReader.hasNext()) {
315                 // Parsing the next event.
316                 if (mDelete == EDelete.ATSTARTMIDDLE) {
317                     /*
318                      * Do not move StAX parser forward if nodes have been
319                      * deleted at the start or in the middle of a subtree.
320                      */
321                     mDelete = EDelete.NODELETE;
322                 } else {
323                     // After an insert or after nodes were equal.
324                     event = mReader.nextEvent();
325                     if (event.isCharacters() && event.asCharacters().isWhiteSpace()) {
326                         continue;
327                     }

330                     assert event != null;
331                     if (firstEvent) {
332                         // Setup start element from StAX parser.
333                         firstEvent = false;
334 
335                         if (event.getEventType() == XMLStreamConstants.START_DOCUMENT) {
336                             while (mReader.hasNext()
337                                 && event.getEventType() != XMLStreamConstants.START_ELEMENT) {
338                                 event = mReader.nextEvent();
339                             }

353                     }
354                 }

357 
358                 switch (event.getEventType()) {
359                 case XMLStreamConstants.START_ELEMENT:
360                     processStartTag(event.asStartElement());
361                     break;
362                 case XMLStreamConstants.CHARACTERS:
363                     sBuilder.append(event.asCharacters().getData());

366                     }
367                     final Characters text = fac.createCharacters(sBuilder.toString().trim());
368                     processCharacters(text);
369                     sBuilder = new StringBuilder();
370                     break;
371                 case XMLStreamConstants.END_ELEMENT:
372                     processEndTag();
373                     break;
374                 default:
375                     // Other nodes which are currently not supported by
376                     // Treetank.
377                 }
378                 // }

415     /**
416      * Process start tag.
417      * 
418      * @param paramElem
419      *            {@link StartElement} currently parsed.
420      * @throws XMLStreamException
421      *             In case of any StAX parsing error.
422      * @throws IOException
423      *             In case of any I/O error.
424      * @throws TTException
425      *             In case of any Treetank error.
426      */
427     private void processStartTag(final StartElement paramElem) throws IOException, XMLStreamException,
428         TTException {
429         assert paramElem != null;
430 
431         // Initialize variables.
432         initializeVars();
433 
434         // Main algorithm to determine if same, insert or a delete has to be
435         // made.
436         algorithm(paramElem);
437 
438         if (mFound && mIsRightSibling) {
439             mDelete = EDelete.ATSTARTMIDDLE;
440             deleteNode();
441         } else if (!mFound) {
442             // Increment levels.
443             mLevelInToShredder++;
444 
445             insertElementNode(paramElem);
446         } else if (mFound) {
447             // Increment levels.
448             mLevelInToShredder++;
449 
450             sameElementNode();
451         }
452     }

454     /**
455      * Process characters.
456      * 
457      * @param paramText
458      *            {@link Characters} currently parsed.
459      * @throws XMLStreamException
460      *             In case of any StAX parsing error.
461      * @throws IOException
462      *             In case of any I/O error.
463      * @throws TTException
464      *             In case of any Treetank error.
465      */
466     private void processCharacters(final Characters paramText) throws IOException, XMLStreamException,
467         TTException {
468         assert paramText != null;
469         // Initialize variables.
470         initializeVars();
471         final String text = paramText.getData().toString();
472         if (!text.isEmpty()) {
473             // Main algorithm to determine if same, insert or a delete has to be
474             // made.
475             algorithm(paramText);
476 
477             if (mFound && mIsRightSibling) {
478                 /*
479                  * Cannot happen because if text node after end tag get's
480                  * deleted it's done already while parsing the end tag. If text
481                  * node should be deleted at the top of a subtree (right after a
482                  * start tag has been parsed) it's done in
483                  * processStartTag(StartElement).
484                  */
485                 // mDelete = EDelete.ATSTARTMIDDLE;
486                 // deleteNode();
487                 throw new AssertionError("");
488             } else if (!mFound) {
489                 insertTextNode(paramText);
490             } else if (mFound) {
491                 sameTextNode();
492             }
493         }
494     }

496     /**
497      * Process end tag.
498      * 
499      * @throws XMLStreamException
500      *             In case of any parsing error.
501      * @throws TTException
502      *             In case anything went wrong while moving/deleting nodes in
503      *             Treetank.
504      */
505     private void processEndTag() throws XMLStreamException, TTException {
506         mLevelInToShredder--;
507 
508         if (mInserted) {
509             mInsertedEndTag = true;
510         }
511 
512         if (mRemovedNode) {
513             mRemovedNode = false;
514         } else {
515 
516             // Move cursor to parent.
517             if (mWtx.getNode().getNodeKey() == mLastNodeKey) {
518                 /*
519                  * An end tag must have been parsed immediately before and it
520                  * must have been an empty element at the end of a subtree, thus
521                  * move this time to parent node.
522                  */
523                 assert mWtx.getNode().hasParent() && mWtx.getNode().getKind() == IConstants.ELEMENT;
524                 mWtx.moveTo(mWtx.getNode().getParentKey());
525             } else {
526                 if (mWtx.getNode().getKind() == IConstants.ELEMENT) {
527                     final ElementNode element = (ElementNode)mWtx.getNode();
528                     if (element.hasFirstChild() && element.hasParent()) {
529                         // It's not an empty element, thus move to parent.
530                         mWtx.moveTo(mWtx.getNode().getParentKey());
531                     }
532                     // } else {
533                     // checkIfLastNode(true);
534                     // }
535                 } else if (((IStructNode)mWtx.getNode()).hasParent()) {
536                     if (((IStructNode)mWtx.getNode()).hasRightSibling()) {
537                         mWtx.moveTo(((IStructNode)mWtx.getNode()).getRightSiblingKey());
538                         /*
539                          * Means next event is an end tag in StAX reader, but
540                          * something different where the Treetank transaction
541                          * points to, which also means it has to be deleted.
542                          */
543                         mKeyMatches = -1;
544                         mDelete = EDelete.ATBOTTOM;
545                         deleteNode();
546                     }
547                     mWtx.moveTo(mWtx.getNode().getParentKey());
548                 }
549 
550             }
551 
552             mLastNodeKey = mWtx.getNode().getNodeKey();
553 
554             // Move cursor to right sibling if it has one.
555             if (((IStructNode)mWtx.getNode()).hasRightSibling()) {
556                 mWtx.moveTo(((IStructNode)mWtx.getNode()).getRightSiblingKey());
557                 mMovedToRightSibling = true;
558 
559                 skipWhitespaces(mReader);
560                 if (mReader.peek().getEventType() == XMLStreamConstants.END_ELEMENT) {
561                     /*
562                      * Means next event is an end tag in StAX reader, but
563                      * something different where the Treetank transaction points
564                      * to, which also means it has to be deleted.
565                      */
566                     mKeyMatches = -1;
567                     mDelete = EDelete.ATBOTTOM;
568                     deleteNode();
569                 }
570             } else {
571                 mMovedToRightSibling = false;
572             }
573         }
574     }

576     /**
577      * Main algorithm to determine if nodes are equal, have to be inserted, or
578      * have to be removed.
579      * 
580      * @param paramEvent
581      *            The currently parsed StAX event.
582      * @throws IOException
583      *             In case the open operation fails (delegated from
584      *             checkDescendants(...)).
585      * @throws XMLStreamException
586      *             In case any StAX parser problem occurs.
587      * @throws TTIOException
588      */
589     private void algorithm(final XMLEvent paramEvent) throws IOException, XMLStreamException, TTIOException {
590         assert paramEvent != null;
591         do {
592             /*
593              * Check if a node in the shreddered file on the same level equals
594              * the current element node.
595              */
596             if (paramEvent.isStartElement()) {
597                 mFound = checkElement(paramEvent.asStartElement());
598             } else if (paramEvent.isCharacters()) {
599                 mFound = checkText(paramEvent.asCharacters());
600             }
601             if (mWtx.getNode().getNodeKey() != mNodeKey) {
602                 mIsRightSibling = true;
603             }
604 
605             mKeyMatches = mWtx.getNode().getNodeKey();
606             //
607             // if (mFound && mIsRightSibling) {
608             // /*
609             // * Root element of next subtree in shreddered file matches
610             // * so check all descendants. If they match the node must be
611             // * inserted.
612             // */
613             // switch (paramEvent.getEventType()) {
614             // case XMLStreamConstants.START_ELEMENT:
615             // mMoved = EMoved.FIRSTNODE;
616             // //mFound = checkDescendants(paramEvent.asStartElement());
617             // mFound = checkDescendants(paramEvent.asStartElement());
618             // break;
619             // case XMLStreamConstants.CHARACTERS:
620             // mFound = checkText(paramEvent.asCharacters());
621             // break;
622             // default:
623             // // throw new
624             // AssertionError("Node type not known or not implemented!");
625             // }
626             // mWtx.moveTo(mKeyMatches);
627             // }
628         } while (!mFound && mWtx.moveTo(((IStructNode)mWtx.getNode()).getRightSiblingKey()));
629         mWtx.moveTo(mNodeKey);
630     }

826     private void insertElementNode(final StartElement paramElement) throws TTException, XMLStreamException {
827         assert paramElement != null;
828         /*
829          * Add node if it's either not found among right siblings (and the
830          * cursor on the shreddered file is on a right sibling) or if it's not
831          * found in the structure and it is a new last right sibling.
832          */
833         mDelete = EDelete.NODELETE;
834         mRemovedNode = false;
835 
836         switch (mInsert) {
837         case ATTOP:
838             // We are at the top of a subtree, no end tag has been parsed
839             // before.

842                 mWtx.moveTo(mWtx.getNode().getParentKey());
843             }
844 
845             // Insert element as first child.
846             addNewElement(EAdd.ASFIRSTCHILD, paramElement);
847             mInsert = EInsert.INTERMEDIATE;
848             break;
849         case INTERMEDIATE:

863             // next node.

865                 mWtx.moveTo(((IStructNode)mWtx.getNode()).getLeftSiblingKey());

869             // as a right sibling.

871                 insertNode = EAdd.ASRIGHTSIBLING;
872             }
873 
874             addNewElement(insertNode, paramElement);
875             break;
876         case ATMIDDLEBOTTOM:
877             // Insert occurs at the middle or end of a subtree.
878 
879             // Move one sibling back.
880             if (mMovedToRightSibling) {
881                 mMovedToRightSibling = false;
882                 mWtx.moveTo(((IStructNode)mWtx.getNode()).getLeftSiblingKey());
883             }
884 
885             // Insert element as right sibling.
886             addNewElement(EAdd.ASRIGHTSIBLING, paramElement);
887             mInsert = EInsert.INTERMEDIATE;
888             break;

891         }

1005     private void deleteNode() throws TTException {

1007          * If found in one of the rightsiblings in the current shreddered
1008          * structure remove all nodes until the transaction points to the found
1009          * node (keyMatches).
1010          */
1011         if (mInserted && !mMovedToRightSibling) {
1012             mInserted = false;
1013             if (((IStructNode)mWtx.getNode()).hasRightSibling()) {
1014                 // Cursor is on the inserted node, so move to right sibling.
1015                 mWtx.moveTo(((IStructNode)mWtx.getNode()).getRightSiblingKey());
1016             }

1020         // checkIfLastNode(true);

1023         // operation.
1024         boolean movedToParent = false;

1027         // deleted.
1028         boolean isLast = false;
1029

1031             if (mWtx.getNode().getNodeKey() != mKeyMatches) {
1032                 final IStructNode node = (IStructNode)mWtx.getNode();
1033                 if (!node.hasRightSibling() && !node.hasLeftSibling()) {
1034                     // if (mDelete == EDelete.ATSTARTMIDDLE) {

1039                     /*

1041                      * moves to the parent after the delete.
1042                      */
1043                     movedToParent = true;
1044                 } else if (!node.hasRightSibling()) {
1045                     // Last node has been reached, which means that the
1046                     // transaction moves to the left sibling.
1047                     isLast = true;
1048                 }
1049 
1050                 mWtx.remove();
1051             }
1052         } while (mWtx.getNode().getNodeKey() != mKeyMatches && !movedToParent && !isLast);

1057                  * Deleted right before an end tag has been parsed, thus don't
1058                  * move transaction to next node in processEndTag().
1059                  */
1060                 mRemovedNode = true;
1061             }
1062             /*
1063              * Treetank transaction has been moved to parent, because all child
1064              * nodes have been deleted, thus to right sibling.
1065              */
1066             mWtx.moveTo(((IStructNode)mWtx.getNode()).getRightSiblingKey());
1067         } else {

1072                      * don't move transaction to next node in processEndTag().
1073                      */

1079                     // parent and right sibling.
