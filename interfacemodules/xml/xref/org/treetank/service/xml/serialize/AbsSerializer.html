42  /**
43   * Class implements main serialization algorithm. Other classes can extend it.
44   * 
45   * @author Johannes Lichtenberger, University of Konstanz
46   * 
47   */
48  abstract class AbsSerializer implements Callable<Void> {
49  
50      /** Treetank session {@link ISession}. */
51      protected final ISession mSession;
52  
53      /** Stack for reading end element. */
54      protected final Stack<Long> mStack;
55  
56      /** Array with versions to print. */
57      protected final long[] mVersions;
58  
59      /** Root node key of subtree to shredder. */
60      protected final long mNodeKey;
61  
62      /**
63       * Constructor.
64       * 
65       * @param paramSession
66       *            {@link ISession}.
67       * @param paramVersions
68       *            versions which should be serialized: -
69       */
70      public AbsSerializer(final ISession paramSession, final long... paramVersions) {
71          this(paramSession, 0, paramVersions);
72      }
73  
74      /**
75       * Constructor.
76       * 
77       * @param paramSession
78       *            {@link ISession}.
79       * @param paramKey
80       *            Key of root node from which to shredder the subtree.
81       * @param paramVersions
82       *            versions which should be serialized: -
83       */
84      public AbsSerializer(final ISession paramSession, final long paramKey, final long... paramVersions) {
85          mStack = new Stack<Long>();
86          mVersions = paramVersions;
87          mSession = paramSession;
88          mNodeKey = paramKey;
89      }
90  
91      /**
92       * Serialize the storage.
93       * 
94       * @return null.
95       * @throws TTException
96       *             if can't call serailzer
97       */
98      public Void call() throws TTException {
99          emitStartDocument();
100 
101         long[] versionsToUse;
102 
103         // if there is one negative number in there, serialize all versions
104         if (mVersions.length == 0) {
105             versionsToUse = new long[] {
106                 mSession.getMostRecentVersion()
107             };
108         } else {
109             if (mVersions.length == 1 && mVersions[0] < 0) {
110                 versionsToUse = null;
111             } else {
112                 versionsToUse = mVersions;
113             }
114         }
115 
116         for (long i = 0; versionsToUse == null ? i < mSession.getMostRecentVersion()
117             : i < versionsToUse.length; i++) {
118 
119             INodeReadTrx rtx =
120                 new NodeReadTrx(mSession.beginPageReadTransaction(versionsToUse == null ? i
121                     : versionsToUse[(int)i]));
122 
123             if (versionsToUse == null || mVersions.length > 1) {
124                 emitStartManualElement(i);
125             }
126 
127             rtx.moveTo(mNodeKey);
128 
129             final AbsAxis descAxis = new DescendantAxis(rtx, true);
130 
131             // Setup primitives.
132             boolean closeElements = false;
133             long key = rtx.getNode().getNodeKey();
134 
135             // Iterate over all nodes of the subtree including self.
136             while (descAxis.hasNext()) {
137                 key = descAxis.next();
138                 IStructNode currentStruc = (IStructNode)rtx.getNode();
139 
140                 // Emit all pending end elements.
141                 if (closeElements) {
142                     while (!mStack.empty() && mStack.peek() != currentStruc.getLeftSiblingKey()) {
143                         rtx.moveTo(mStack.pop());
144                         emitEndElement(rtx);
145                         rtx.moveTo(key);
146                     }
147                     if (!mStack.empty()) {
148                         rtx.moveTo(mStack.pop());
149                         emitEndElement(rtx);
150                     }
151                     rtx.moveTo(key);
152                     closeElements = false;
153                 }
154 
155                 // Emit node.
156                 emitStartElement(rtx);
157 
158                 // Push end element to stack if we are a start element with
159                 // children.
160                 if (currentStruc.getKind() == IConstants.ELEMENT && currentStruc.hasFirstChild()) {
161                     mStack.push(rtx.getNode().getNodeKey());
162                 }
163 
164                 // Remember to emit all pending end elements from stack if
165                 // required.
166                 if (!currentStruc.hasFirstChild() && !currentStruc.hasRightSibling()) {
167                     closeElements = true;
168                 }
169 
170             }
171 
172             // Finally emit all pending end elements.
173             while (!mStack.empty()) {
174                 rtx.moveTo(mStack.pop());
175                 emitEndElement(rtx);
176             }
177 
178             if (versionsToUse == null || mVersions.length > 1) {
179                 emitEndManualElement(i);
180             }
181         }
182         emitEndDocument();
183 
184         return null;
185     }

188     protected abstract void emitStartDocument();

191      * Emit start tag.
192      * 
193      * @param paramRTX
194      *            Treetank reading transaction {@link IReadTransaction}.

196     protected abstract void emitStartElement(final INodeReadTrx paramRTX);

199      * Emit end tag.
200      *

204     protected abstract void emitEndElement(final INodeReadTrx paramRTX);

213

216      *

220     protected abstract void emitEndManualElement(final long paramVersion);
221 
222     /** Emit end document. */
223     protected abstract void emitEndDocument();
