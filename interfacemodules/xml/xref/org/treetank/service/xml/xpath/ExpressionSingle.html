45  /**
46   * <h1>ExpresseionSingle</h1>
47   * <p>
48   * This class builds an execution chain to execute a XPath query. All added axis are build together by using
49   * NestedAxis.
50   */
51  public class ExpressionSingle {
52  
53      /**
54       * Counts the number of added axis. This is used to handle the special
55       * behavior for the first and second axis that are added.
56       */
57      private int mNumber;
58  
59      /** The first added axis has to be stored till a second one is added. */
60      private AbsAxis mFirstAxis;
61  
62      /** Contains the execution chain consisting of nested NestedAxis. */
63      private AbsAxis mExpr;
64  
65      /** Current ordering state. */
66      private OrdState mOrd;
67  
68      /** Current duplicate state. */
69      private DupState mDup;
70  
71      /** Private IReadTrx for generating new axis. */
72      private final INodeReadTrx mRtx;
73  
74      /**
75       * Constructor. Initializes the internal state.
76       */
77      public ExpressionSingle(final INodeReadTrx pRtx) {
78  
79          mNumber = 0;
80  
81          mOrd = OrdState.MAX1;
82          mOrd.init();
83          mDup = DupState.MAX1;
84          mRtx = pRtx;
85  
86      }
87  
88      /**
89       * Adds a new Axis to the expression chain. The first axis that is added has
90       * to be stored till a second axis is added. When the second axis is added,
91       * it is nested with the first one and builds the execution chain.
92       * 
93       * @param mAx
94       *            ach The axis to add.
95       */
96      public void add(final AbsAxis mAx) {
97          AbsAxis axis = mAx;
98  
99          if (isDupOrd(axis)) {
100             axis = new DupFilterAxis(mRtx, axis);
101             DupState.nodup = true;
102         }
103 
104         switch (mNumber) {
105         case 0:
106             mFirstAxis = axis;
107             mNumber++;
108             break;
109         case 1:
110             mExpr = new NestedAxis(mFirstAxis, axis, mRtx);
111             mNumber++;
112             break;
113         default:
114             final AbsAxis cache = mExpr;
115             mExpr = new NestedAxis(cache, axis, mRtx);
116         }
117 
118     }
119 
120     /**
121      * Returns a chain to execute the query. If there is only one axis added,
122      * the chain was not build yet, so only this axis is returned.
123      * 
124      * @return The query execution chain
125      */
126     public AbsAxis getExpr() {
127 
128         return (mNumber == 1) ? mFirstAxis : mExpr;
129     }
130 
131     /**
132      * Returns the number of axis in this expression.
133      * 
134      * @return size of the expression
135      */
136     public int getSize() {
137 
138         return mNumber;
139     }
140 
141     /**
142      * Determines for a given string representation of an axis, whether this
143      * axis leads to duplicates in the result sequence or not. Furthermore it
144      * determines the new state for the order state that specifies, if the
145      * result sequence is in document order. This method is implemented
146      * according to the automata in [Hidders, J., Michiels, P., "Avoiding
147      * Unnecessary Ordering Operations in XPath", 2003]
148      * 
149      * @param ax
150      *            name of the current axis
151      * @return true, if expression is still duplicate free
152      */
153     public boolean isDupOrd(final AbsAxis ax) {
154 
155         AbsAxis axis = ax;
156 
157         while (axis instanceof FilterAxis) {
158             axis = ((FilterAxis)axis).getAxis();
159         }
160 
161         if (axis instanceof UnionAxis) {
162             mOrd = mOrd.updateOrdUnion();
163             mDup = mDup.updateUnion();
164 
165         } else if (axis instanceof ChildAxis) {
166             mOrd = mOrd.updateOrdChild();
167             mDup = mDup.updateDupChild();
168 
169         } else if (axis instanceof ParentAxis) {
170 
171             mOrd = mOrd.updateOrdParent();
172             mDup = mDup.updateDupParent();
173 
174         } else if (axis instanceof DescendantAxis) {
175 
176             mOrd = mOrd.updateOrdDesc();
177             mDup = mDup.updateDupDesc();
178 
179         } else if (axis instanceof AncestorAxis) {
180 
181             mOrd = mOrd.updateOrdAncestor();
182             mDup = mDup.updateDupAncestor();
183 
184         } else if (axis instanceof FollowingAxis || axis instanceof PrecedingAxis) {
185 
186             mOrd = mOrd.updateOrdFollPre();
187             mDup = mDup.updateDupFollPre();
188 
189         } else if (axis instanceof FollowingSiblingAxis || axis instanceof PrecedingSiblingAxis) {
190 
191             mOrd = mOrd.updateOrdFollPreSib();
192             mDup = mDup.updateDupFollPreSib();
193         }
194 
195         return !DupState.nodup;
196     }
197 
198     /**
199      * @return true, if the result is in document order
200      */
201     public boolean isOrdered() {
202 
203         // the result sequence is unordered, if the order rank is greater than
204         // zero
205         // or the order state is in state UNORD
206         // return (ord != OrdState.UNORD && ord.mOrdRank == 0);
207         return (mOrd != OrdState.UNORD && OrdState.mOrdRank == 0);
208     }
